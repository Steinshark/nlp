from tkinter import Tk, ttk, scrolledtext, Menu

root = Tk()
root.geometry('800x600')
root.rowconfigure(0,weight=50)
root.rowconfigure(1,weight=50)
root.columnconfigure(0,weight=20)
root.columnconfigure(1,weight=80)
root.title("Your App Name")

#Make working sidebar
sidebar     = ttk.Frame(root)
sidebar.columnconfigure(0,weight=1)
sidebar.rowconfigure(0,weight=1)
sidebar.rowconfigure(1,weight=1)
sidebar.rowconfigure(2,weight=18)
sidebar.grid(row=0,column=0,rowspan=2,sticky='NS')

sbutton = ttk.Button(sidebar)

def open_inventory_mode():
   
    #Destroy old items 
    old_items    = [sidebar.children[item] for item in sidebar.children]
    for item in old_items:
        item.destroy()

    #Create new items 
    add_item_button     = ttk.Button(sidebar,text="Add Item")
    add_item_button.grid(row=0,column=0,sticky='EW')

    add_rand_button     = ttk.Button(sidebar,text="Add Rand")
    add_rand_button.grid(row=2,column=0,sticky='EW')

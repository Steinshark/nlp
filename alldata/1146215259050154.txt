unsorted_list   = [45,23,67,100,54]
sorted_list     = [] 

while len(unsorted_list) > 0:

    #Find the top value in the unsorted_list
    max_value   = unsorted_list[0]


    #Add it to the sorted_list
    sorted_list.append(max_value)

    #Remove from the unsorted_list 
    unsorted_list.remove(max_value)

print(sorted_list)
